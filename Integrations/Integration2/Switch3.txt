enable
  configure terminal
    vlan 10
      name Serveur
    vlan 20
      name Charles
    exit
    interface f0/1
      no shutdown
      switchport mode access
      switchport access vlan 10
    interface f0/2
      no shutdown
      switchport mode access
      switchport access vlan 20
    interface f0/3
      no shutdown
      switchport mode trunk
      switchport trunk allowed vlan 10,20
    interface range f0/4 - 24
      shutdown
    exit
  exit
  copy running-config startup-config
