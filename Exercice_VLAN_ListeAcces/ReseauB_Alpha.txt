enable
  configure terminal
    vlan 10
      name V10
    vlan 20
      name V20
    exit
    interface f0/1
      no shutdown
      switchport mode access
      switchport access vlan 10
    interface f0/2
      no shutdown
      switchport mode access
      switchport access vlan 20
    interface f0/3
      no shutdown
      switchport mode access
      switchport access vlan 20
    interface f0/4
      no shutdown
      switchport mode trunk
      switchport trunk allowed vlan all
    interface range f0/5 - f0/24
      no switchport mode
      shutdown
    exit
  exit
  copy running-config startup-config

exit
